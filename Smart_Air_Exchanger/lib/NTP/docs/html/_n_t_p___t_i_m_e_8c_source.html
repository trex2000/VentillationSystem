368 void NTP_v_tryConnect(void)
369 {
370  // If the connection was succesful and the time was retrieved correctly, increment refresh time count for
371  // the next time synchronisation, else reset the refresh counter
372  if (!NTP_ub_connectionError)
373  {
374  NTP_i_refreshTime++;
375  }
376  else

382  {

384  NTP_ub_connectionError = NTP_CON_ERROR;
